47def evolve(grid): 

48 x = len(grid) 

49 y = len(grid[0]) 

50 new_grid = make_empty_grid(x, y) 

51 for r in range(x): 

52 for c in range(y): 

53 cell = grid[r][c] 

54 neighbours = count_neighbours(grid, (r, c)) 

55 new_grid[r][c] = 1 if evolve_cell(cell, neighbours) else 0
